1. Lendo dados

Podemos ter um arquivo nomes.txt: 

Fulano
Beltrano
Ciclano

E ler cada linha imprimindo: 

Nome 1: Fulano
Nome 2: Beltrano
Nome 3: Ciclano

Para isso, precisamos:

  1. abrir o arquivo para leitura.
  2. ler cada linha.
  3. imprimir a linha lida.
  4. fechar o arquivo.

1.1. Abrir arquivo para leitura

Usamos fopen() passando o nome do arquivo e o modo leitura: 

fopen("nomes.txt", "r");

1.2. Ler cada linha

Usamos fgets() para ler um conjunto de caracteres até '\n', passamos o destino onde será colocados os caracteres, a quantidade máxima de caracteres que poderão ser lidos e um ponteiro para o arquivo que será lido: 

fgets(destino, maximo, parquivo);

A função fgets() retorna um ponteiro para o buffer lido ou NULL se houver erro.

1.3. Imprimir cada linha

Como a função fgets() salva os caracteres em um vetor destino podemos imprimir o vetor usando: 

printf("%s", destino);

1.4. Fechar o arquivo

Para fechar o arquivo após o uso, a função fclose() é empregada assim: 

fclose(parquivo);

1.5. Exemplo

#include <stdio.h>

int main(void) {
  FILE *pArquivo = NULL;
  pArquivo = fopen("nomes.txt", "r");

  char destino[80];
  fgets(destino, 80, pArquivo);

  printf("%s", destino);

  fclose(pArquivo);
  return 0;
}

2. Escrevendo dados

Para receber um dado e gravá-lo em um arquivo devemos:

  1. receber o dado.
  2. abrir o arquivo para escrita.
  3. escrever o dado no arquivo.
  4. fechar o arquivo.

2.1. Recebendo o dado

Usamos scanf() para receber dados do usuário: 

scanf("%s", nome);

2.2. Abrir arquivo para escrita

Usamos a mesma fopen() com o modo de escrita:

"w"
escrever apagando o conteúdo anterior.
"a"
escrever no fim do arquivo.
fopen("nomes.txt", "a");

2.3. Escrever dado

A função fputs() escreve uma string em um determinado arquivo: 

fputs( nome, parquivo);

Ela retorna um valor não negativo se for tudo bem na escrita ou EOF caso ocorra um erro de escrita.

2.4. Fechar o arquivo

Para fechar o arquivo após a escrita, a mesma função fclose() é usada: 

fclose(parquivo);

2.5. Exemplo

#include <stdio.h>

int main(void) {
    // Lendo dado
    char nome[20];
    printf("Nome: ");
    scanf("%19s", nome);

    // abrindo arquivo
    FILE *pArquivo = NULL;
    pArquivo = fopen("nomes.txt", "a");

    // escrevendo no arquivo
    fputs(nome, pArquivo);

    // fechando o arquivo
    fclose(pArquivo);

    return 0;
}

2.6. Dados formatados

Para escrever formatando os dados use fprintf(): 

fprintf(parquivo, "%s\n", nome);

3. Struct

Structs formam tipos de dados: 

struct funcionario {
  char nome[20];
  int idade;
  double salario;
};

Para declarar variáveis usamos a tag pessoa: 

struct funcionario primeiro;

Um ponteiro para um struct funcionario fica assim: 

struct funcionario *pFunc;

Vários funcionários podem ser armazenados em arrays: 

struct funcionario empresa[20];

A declaração pode vir após a definição: 

struct funcionario {
  char nome[20];
  int idade;
  double salario;
} primeiro;

É útil para quando não vamos usar mais a struct e podemos deixá-lá anônima: 

struct {
  char nome[20];
  int idade;
  double salario;
} primeiro;

3.1. Usando structs

Vamos gravar um struct em arquivo usando fwrite(): 

#include <stdio.h>

struct funcionario {
  char nome[20];
  int idade;
  double salario;
};

int main(void) {
  struct funcionario unico;

  printf("Digite o nome: ");
  scanf("%19s", unico.nome);

  printf("Digite a idade: ");
  scanf("%d", &unico.idade);

  printf("Digite o salario: ");
  scanf("%lf", &unico.salario);

  FILE *pArquivo;
  pArquivo = fopen("empresa.txt", "wb");

  fwrite(&unico, sizeof(struct funcionario), 1, pArquivo);

  fclose(pArquivo);

  return 0;
}

A sintaxe de fwrite é: 

fwrite(origem, tamanho, quantidade, arquivo);
origem
ponteiro para o dado que será gravado
tamanho
tamanho de um dado gravado no arquivo.
quantidade
quantidade de dados que serão gravados.
arquivo
ponteiro para o arquivo aberto.

3.2. Lendo dados

Usando fread() lemos dados binários: 

#include <stdio.h>

struct funcionario {
  char nome[20];
  int idade;
  double salario;
};

int main(void) {
  struct funcionario retorno;

  FILE *pArquivo;
  pArquivo = fopen("empresa.txt", "rb");

  fread(&retorno, sizeof(struct funcionario), 1, pArquivo);

  fclose(pArquivo);

  printf("Nome: %s\n", retorno.nome);
  printf("Idade: %d\n", retorno.idade);
  printf("Salário: %f\n", retorno.salario);

  return 0;
}

A sintaxe é: 

fread(destino, tamanho, quantidade, arquivo);
destino
ponteiro para onde vão os dados lidos.
tamanho
tamanho de um dado gravado no arquivo.
quantidade
quantidade de dados que serão lidos.
arquivo
ponteiro para o arquivo aberto.
